## frps.ini 服务器端

{% highlight shell %}
bind_port = 1234
vhost_http_port = 5678
privilege_token = xxx
privilege_mode = true
{% endhighlight %}

## frpc.ini 客户端

{% highlight shell %}
[common]
# 服务器地址
server_addr = your_ip
# 服务器端口
server_port = 1234
# 服务器连接凭证
privilege_token = xxx
[port1]
type = http
local_port = 6000
remote_port = 5678
custom_domains = your_domain
{% endhighlight %}

## 用systemctl来控制启动

* 新建一个 /lib/systemd/system/frps.service

{% highlight shell %}
[Unit]
Description=frps service
After=network.target syslog.target
Wants=network.target
[Service]
Type=simple
#启动服务的命令（此处写你的frps的实际安装目录）
ExecStart=/your/path/frps -c /your/path/frps.ini
[Install]
WantedBy=multi-user.target
{% endhighlight %}

* 启动

{% highlight shell %}
# systemctl start frps
{% endhighlight %}

* 自启

{% highlight shell %}
# systemctl enable frps
{% endhighlight %}
